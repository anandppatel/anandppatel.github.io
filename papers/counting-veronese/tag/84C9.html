Lemma 5.1. (84C9)

% We have: %

  1. $\left[\Thin\right] = c_{1}\left(\O(1)^{[3]}\right)$. \label{thineq}\\ %
  2. $H = c_{1}\left(\O(2)^{[3]}\right) - c_{1}\left(\O(1)^{[3]}\right)$. \label{Heq}\\ %
  3. $\left[\Lin(p)\right] = c_2 \left( \O(1)^{[3]} \right)$.\label{linpeq}\\ %
  4. $2 \left[\Inc(p) \right] + 2 \left[\Lin(p)\right] + \left[ \Inc^{*}(\ell^{*}) \right] + \left[ \Lin^{*}(\ell^{*})\right] = c_{1}^{2}\left(\O(2)^{[3]}\right)-c_{2}\left(\O(2)^{[3]}\right)$.\label{inclineq}\\ %
  5. $\left[\Inc^{*}(\ell^{*})\right] +2 \left[\Lin^{*}(\ell^{*})\right]+\left[\Lin(p)\right]+H\cdot H^{*} = c_{2}\left(\O(2)^{[3]}\right)$. %
